public class Test563 {

    public static void main(String[] args) {
        // TODO Auto-generated method stub
      TreeNode f = new TreeNode(2);
      TreeNode r = new TreeNode(1);
      r.left = f;
      System.out.println(findTilt(r));
    }

    static int ans = 0;

    public static  int findTilt(TreeNode root) {
        ans = 0;
        calSum(root);
        return ans;
    }
    
    public static int  calSum(TreeNode root)
    {
        if (root == null)
            return 0;
        int sum_left = calSum(root.left);
        int sum_right = calSum(root.right);
        ans +=Math.abs(sum_left - sum_right);
        return sum_left + root.val + sum_right;
    }

}
